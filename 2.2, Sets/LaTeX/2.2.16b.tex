	\begin{theorem*}[2.2.16b]
		Let A and B bet sets. $A \subseteq (A \cup B)$
	\end{theorem*}
	
	\begin{proof}
		All of the elements in A are a subset of $A \cup B$ by the definition of set union. Therefore it trivially follows that $A \subseteq (A \cup B)$.
	\end{proof}
