\begin{document}
	
	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.10}
		
		\caption{compute $x^{n}$}\label{power}
		
		\begin{algorithmic}[1]
			
			\Procedure{power}{$n$: integer; $x$: real number}
			
			\State $exponent \gets |n|$
			\State $product \gets 1$
			
			\While{\texttt{$exponent > 0$}} \Comment {multiply $x$ by itself $|n|$ times}
				%\State $x \gets x \times x$
				\State $product \gets product \times x$
				\State $exponent \minuseq 1$
			\EndWhile
			\If {$n < 0$} \Comment{n is negative so get the inverse}
				%\State $x \gets \frac{1}{x}$
				\State $product \gets \frac{1}{product}$
			\EndIf
			\State \textbf{return} $product$ \Comment{this $product$ is $x^{n}$}
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
