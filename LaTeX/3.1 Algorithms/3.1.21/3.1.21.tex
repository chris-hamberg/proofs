	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.21}
		
		\caption{Sort the first three terms in a sequence of integers}
		
		\begin{algorithmic}[1]
			
			\Procedure{partial sort}{$a_1$, $a_2$, $\dots$, $a_n$: sequence of integers}
			
			\For {$i = 1$ to $2$}\Comment {Bubble Sort $a_1$, $a_2$, $a_3$ and then stop}
				\For {$j = 1$ to $2$}
					\If {$a_j > a_{j+1}$}
						\State swap $a_j$ and $a_{j+1}$
					\EndIf
				\EndFor
			\EndFor
			
			\State \textbf{return} $a_1$, $a_2$, $\dots$, $a_n$
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
