	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.22}
		
		\caption{Find the longest English word in an English sentence.}
		
		\begin{algorithmic}[1]
			
			\Procedure{longest word}{$a_1$, $a_2$, $\dots$, $a_n$: string of characters}
			
			\State $word$ $length \gets 0$
			\State $maximum$ $size \gets 0$
			\State $maximum$ $index \gets 0$
			\State $a_{n+1} \gets "$ $"$
			
			\For {$i = 1$ to $(n+1)$}
				\If {$a_i \ne "$ $"$} \Comment {It's in the middle of a word.}
					\State $word$ $length \pluseq 1$
				\Else \Comment{End of word has been reached.}
					\State $index \gets (i - word$ $length)$ \Comment {Record start index for word.}
					\If {$word$ $length > maximum$ $size$}
						\State $maximum$ $size \gets word$ $length$
						\State $maximum$ $index \gets index$
					\EndIf
					\State $word$ $length \gets 0$
				\EndIf
			\EndFor
			
			\State \textbf{return} $maximum$ $index$
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
