	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.31}
		
		\caption{Find the first duplicate term in a sequence of integers}
		
		\begin{algorithmic}[1]
			
			\Procedure{duplicate term}{$a_1$, $a_2$, $\dots$, $a_n$: sequence of integers}
			
			\For {$index = 2$ to $n$}
				\For {$i = 1$ to $(index-1)$}
					\If {$a_i = a_{index}$}\Comment {$a_{index}$ is the first duplicate term.} 
						\State \textbf{return} $a_{index}, index$
					\EndIf
				\EndFor
			\EndFor
			
			\State \textbf{return} $\bot$ \Comment{No duplicate terms exist.}
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
