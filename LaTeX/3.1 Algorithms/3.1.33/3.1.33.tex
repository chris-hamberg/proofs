	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.33}
		
		\caption{Find the first term in a sequence of positive integers where its predecessor is the greater value.}
		
		\begin{algorithmic}[1]
			
			\Procedure{disordered}{$a_1$, $a_2$, $\dots$, $a_n$: sequence of positive integers}
			
			\For {$i = 1$ to $n-1$}
				\If {$a_i > a_{i+1}$}
					\State \textbf{return} $a_{i+1}, i+1$ \Comment {returns (term, index)}
				\EndIf
			\EndFor
			
			\State \textbf{return} $\bot$ \Comment {All terms are in ascending order.}
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
