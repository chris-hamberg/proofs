		\renewcommand{\thealgorithm}{3.1.44}
		
		\caption{List insertion based on binary search for a sorted list.}
		
		\begin{algorithmic}[1]
			
			\Procedure{binary insert}{$x$: object; $a_1$, $a_2$, $\dots$, $a_n$: sorted list of objects}
			
			\State $i \gets 1$
			\State $j \gets n$
			
			\While {$i < j$}\Comment {Binary search.}
				\State $m \gets \floor{\frac{(i+j)}{2}}$
				\If {$x > a_m$} 
					\State $i \gets m+1$
				\Else
					\State $j \gets m$
				\EndIf
			\EndWhile \Comment {End binary search.}
			
			\For {$k = (n+1)$ to $(i-1)$} \Comment {Make room for $x$.}
				\State $a_k \gets a_{k-1}$
			\EndFor
			
			\If {$i > n$} \Comment {$x$ is the greatest element.}
				\State $a_{i+1} \gets x$
			\Else \Comment {The index for $x$ is less than $n+1$}
				\State $a_i \gets x$
			\EndIf
			
			\State \textbf{return} $a_1$, $a_2$, $\dots$, $a_{n+1}$
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
