	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.11}
		
		\caption{swap $x$ and $y$}\label{swap}
		
		\begin{algorithmic}[1]
			
			\Procedure{swap}{$x$: object; $y$: object}
			
			\State $temp \gets x$
			\State $x \gets y$
			\State $y \gets temp$

			\State \textbf{return} $x, y$
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
