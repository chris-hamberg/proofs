	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.15}
		
		\caption{insert an integer $x$ at the correct index position in a list of integers in increasing order}
		
		\begin{algorithmic}[1]
			
			\Procedure{insert}{$x$: integer; $a_1, a_2, \dots, a_n$: integers in increasing \newline order}
			\If {$a_n \le x$}
				\State $a_{n+1} \gets x$
			\Else
				\State $i \gets n$
				\While {$i$ and $(x < a_i)$} \Comment {make room for x}
					\State $a_{i+1} \gets a_i$
					\State $i \minuseq 1$
				\EndWhile
				\State $a_{i+1} \gets x$
			\EndIf

			\State \textbf{return} $a_1, a_2, \dots, a_{n+1}$
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
