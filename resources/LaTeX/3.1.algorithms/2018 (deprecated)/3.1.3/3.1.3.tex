\def\NoNumber#1{{\def\alglinenumber##1{}\State#1}\addtocounter{ALG@line}{-1}}

\newcommand{\pluseq}{\mathrel{+}=}

\begin{document}
	
	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.3}
		
		\caption{calculate sum of integers in a list}
		
		\begin{algorithmic}[1]
			
			\Procedure{sum}{$a_1, a_2, \dots, a_n$: list of integers}
			\State $total \gets 0$
			\For{\texttt{i = 1 to n}}
				\State\texttt{$total \pluseq a_i$}
			\EndFor
			\State \textbf{return} $total$ \Comment{$total = \sum_{i=1}^{n} a_i$}
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
