\usepackage{algorithmicx,algorithm,algpseudocode}

\def\NoNumber#1{{\def\alglinenumber##1{}\State#1}\addtocounter{ALG@line}{-1}}

\newcommand{\pluseq}{\mathrel{+}=}

\begin{document}
	
	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.6}
		
		\caption{count negative list entries}
		
		\begin{algorithmic}[1]
			
			\Procedure{negatives}{$a_1, a_2, \dots, a_n$: list of integers}
			\State $count \gets 0$
			\For{\texttt{i = 1 to n}}
				\If {$a_i < 0$}
					\State $count \pluseq 1$
				\EndIf
			\EndFor
			\State \textbf{return} $count$ 
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
