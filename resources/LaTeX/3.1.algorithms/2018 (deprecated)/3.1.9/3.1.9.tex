	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.9}
		
		\caption{determine whether a string of characters is a palindrome}
		
		\begin{algorithmic}[1]
			
			\Procedure{palindrome}{$a_1, a_2, \dots, a_n$: string of characters}
			\For{\texttt{i = 1 to $\floor{\frac{n}{2}}$}}
				\If {$a_i \ne a_{(n+1)-i}$}
					\State \textbf{return} $\bot$
				\EndIf
			\EndFor
			\State \textbf{return} $True$ \Comment{all of the characters matched}
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
