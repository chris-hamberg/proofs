	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.16}
		
		\caption{Find the smallest number from a finite sequence of natural numbers.}
		
		\begin{algorithmic}[1]
			
			\Procedure{min}{$a_1$, $a_2$, $\dots$, $a_n$: finite sequence of natural numbers}
			\State $minimum \gets a_1$
			\For {$i = 2$ to $n$}
				\If {$a_i < minimum$}
					\State $minimum \gets a_i$
			    \EndIf
			\EndFor
			\State \textbf{return} $minimum$ 
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
