	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.23}
		
		\caption{Determine whether a function from a finite set of integers to a finite set of integers is surjective.}
		
		\begin{algorithmic}[1]
			
			\Procedure{surjection}{$f$: function; $A$: finite set of integers; $B$: finite set of integers}
			
			\For {$b \in B$}
				\For {$a \in A$}
					\If {$b = f(a)$}
						\State $break$ \Comment {So far so good.}
					\Else
						\State \textbf{return} $\bot$ \Comment {$\lnot \forall b \exists a (f(a) = b)$}
					\EndIf
				\EndFor
			\EndFor
			\State \textbf{return} $True$
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
