	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.29}
		
		\caption{Find a mode from a list of nondecreasing integers.}
		
		\begin{algorithmic}[1]
			
			\Procedure{mode}{$a_1$, $a_2$, $\dots$, $a_n$: list of nondecreasing integers}
			
			\State $mode \gets \bot$
			\State $count \gets 1$
			\State $frequency \gets 1$
			\For {$i = 1$ to $n-1$}
				\IfThenElse {$a_i \ne a_{i+1}$} {$count = 1$} {$count \pluseq 1$}
				\If {$count > frequency$} \Comment {Found new sublist mode.}
					\State $mode \gets a_i$
					\State $frequency \gets count$ \Comment {Set mode in memory.}
				\EndIf
			\EndFor
			\State \textbf{return} $mode$
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
