	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.32}
		
		\caption{Find all terms of a finite sequence of integers where the term is greater than the sum of its preceding terms.}
		
		\begin{algorithmic}[1]
			
			\Procedure{large terms}{$a_1$, $a_2$, $\dots$, $a_n$: finite sequence of integers}
			
			
			\State $qualified \gets \emptyset$
			
			\For {$i = 2$ to $n$}
				\If { $(\sum_{j=1}^{i-1} a_j) < a_i$}
					\State $qualified \gets qualified \cup \{a_i\}$
				\EndIf
			\EndFor
			
			\State \textbf{return} $qualified$
			
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
