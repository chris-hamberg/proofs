	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.43}
		
		\caption{List insertion based on linear search for a sorted list.}
		
		\begin{algorithmic}[1]
			
			\Procedure{linear insert}{$x$: object; $a_1$, $a_2$, $\dots$, $a_n$: sorted list of objects}
			
			\State $i \gets 1$
			
			\While {$(i \le n)$ and $(x > a_i)$} \Comment {Find the index for x}
				\State $i \pluseq 1$
			\EndWhile
			
			\If {$x \le a_i$} \Comment {Make room for x}
				\For {$k = (n+1)$ to $(i-1)$}
					\State $a_k \gets a_{k-1}$
				\EndFor
			\EndIf
			
			\State {$a_i \gets x$}
			
			\State \textbf{return} $a_1$, $a_2$, $\dots$, $a_{n+1}$
			
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
