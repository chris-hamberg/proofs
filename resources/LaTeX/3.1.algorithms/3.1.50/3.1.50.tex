	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.50}
		
		\caption{Insertion sort using the reversed linear search technique.}
		
		\begin{algorithmic}[1]
			
			\Procedure{reversed insertion sort}{$a_1$, $a_2$, $\dots$, $a_n$: list of comparable entries}
			
			\For {$j = 2$ to $n$}
				\State $i \gets j$
				\While {$(i \ne 1)$ and $(a_{i-1} > a_j)$} \Comment {RLS starting from $a_{j-1}$.}
					\State $i \minuseq 1$
				\EndWhile \Comment {End linear search.}
				\State $element \gets a_j$ \Comment {Insertion.}
				\For {$k = 0$ to $(j-i-1)$}
					\State $a_{(j-k)} \gets a_{(j-k-1)}$
				\EndFor
				\State $a_i \gets element$ \Comment {End insertion.}
			\EndFor
			\State \textbf{return} $a_1$, $a_2$, $\dots$, $a_n$
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
