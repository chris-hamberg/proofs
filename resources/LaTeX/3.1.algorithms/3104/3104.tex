\documentclass[preview]{standalone}
\usepackage{xcolor}
\usepackage[noend]{algpseudocode}

\begin{document}
    
    \begin{algorithmic}[0]
        
        \Statex \color{white} x \color{black}
        \Function{max difference}{$\rho_1, \rho_2, \dots, \rho_\epsilon$: list of integers}
        \State $\mu \gets 0$
        \For{$\iota = 2, \epsilon$}
            \State $\sigma = \rho_\iota - \rho_{\iota-1}$
            \If {$\mu < \sigma$}
                \State $\mu = \sigma$
            \EndIf
        \EndFor
        \State \Return{$\mu$}
        \EndFunction

    \end{algorithmic}

\end{document}
