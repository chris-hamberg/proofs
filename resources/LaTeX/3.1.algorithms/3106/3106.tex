\begin{center}
    \text{\emph{Count negative valued list entry occurrences.}}
\end{center}

\begin{algorithmic}[0]
    \Statex \color{white} x \color{black}
    \Function{negatives}{$\rho_1, \rho_2, \dots, \rho_\epsilon$: list of integers}
    \State $\phi \gets 0$
    \For{$\iota = 1, \epsilon$}
        \If {$\rho_\iota < 0$}
            \State $\phi \pluseq 1$
        \EndIf
    \EndFor
    \State \Return{$\phi$}
    \EndFunction
    
\end{algorithmic}
