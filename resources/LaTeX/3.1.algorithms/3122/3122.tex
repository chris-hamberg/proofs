	\begin{center}
		\text{\emph{Find the longest English word in an English sentence.}}
		\vspace{0.5\baselineskip}
	\end{center}

	\begin{algorithmic}[0]
			
		\Function{longest word}{$\rho_1$, $\rho_2$, $\dots$, $\rho_\lambda$: string of characters}
			
		\State $\omega \gets 0$ 
		\color{lightgray} \Comment {word length} \color{black}
		\State $\Omega \gets 0$
		\color{lightgray} \Comment {max word size} \color{black}
		\State $\Phi \gets 0$
		\color{lightgray} \Comment {max size index} \color{black}
		\State $\rho_{\lambda + 1} \gets "$ $"$
			
		\For {$\iota = 1, \langle \lambda + 1 \rangle$}
			\If {$\rho_\iota \ne "$ $"$} 
			\color{lightgray} \Comment {It's in the middle of a word.} \color{black}
				\State $\omega \pluseq 1$
			\Else 
			\color{lightgray} \Comment{End of word has been reached.} \color{black}
				\State $\Delta \gets \langle \iota - \omega \rangle$ 
				\color{lightgray} \Comment {Record start index for word.} \color{black}
				\If {$\omega > \Omega$}
					\State $\Omega \gets \omega$
					\State $\Phi \gets \Delta$
				\EndIf
				\State $\omega \gets 0$
			\EndIf
		\EndFor
			
		\State \Return{$\Phi$}
		\EndFunction
			
	\end{algorithmic}
