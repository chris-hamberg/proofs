	\begin{algorithm}

		\renewcommand{\thealgorithm}{3.1.5}
		
		\caption{find duplicate list entries}
		
		\begin{algorithmic}[1]
			
			\Procedure{duplicates}{$a_1, a_2, \dots, a_n$: integers in nondecreasing order}
			\State $positives \gets set()$
			\For{\texttt{i = 1 to n-1}}
				\If {$a_i = a_{i+1}$}
					\State $positives \gets positives \cup \{a_{i+1}\}$
				\EndIf
			\EndFor
			\State \textbf{return} $positives$ 
			\EndProcedure
			
		\end{algorithmic}

	\end{algorithm}
